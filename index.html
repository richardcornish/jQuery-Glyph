jQuery Glyph demos

Code

<!-- jQuery from Google, latest 1.x.x version, minified for production -->
<script src="http://ajax.googleapis.com/ajax/libs/jquery/1/jquery.min.js"></script>

<!-- jQuery Glyph, minified for production -->
<script src="jquery.glyph.min.js"></script>

<!-- Call jQuery Glyph on any selector with preferred options -->
<script>
$(document).ready(function(){

    // Demo 1: Default entity
    $('.article-1').glyph();

    // Demo 2: Red snowman
    $('.article-2').glyph({ entity: '&#9731;', color: '#f00', size: '200%', left: '0', top: '17px' });

    // Demo 3: Apple favicon
    $('.article-3').glyph({ domain: 'apple.com' });

    // Demo 4: White House favicon
    $('.article-4').glyph({ favicon: 'http://www.whitehouse.gov/sites/default/themes/whitehouse/favicon.ico' });

});
</script>
