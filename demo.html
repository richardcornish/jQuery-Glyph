jQuery Glyph demos

Code

<!-- jQuery from Google, latest 1.x.x version, minified for production -->
<script type="text/javascript" src="http://ajax.googleapis.com/ajax/libs/jquery/1/jquery.min.js"></script>

<!-- jQuery Glyph, minified for production -->
<script type="text/javascript" src="jquery.glyph.min.js"></script>

<!-- Call jQuery Glyph on any selector with preferred options -->
<script type="text/javascript">
$(document).ready(function(){
    
    // Demo 1: Default entity
    $('#demo-1').glyph();
    
    // Demo 2: Red snowman
    $('#demo-2').glyph({ entity: '&#9731;', color: '#f00', size: '200%', left: '0', top: '17px' });
    
    // Demo 3: Apple favicon
    $('#demo-3').glyph({ domain: 'barackobama.com' });
    
    // Demo 4: White House favicon
    $('#demo-4').glyph({ favicon: 'http://www.whitehouse.gov/sites/default/themes/whitehouse/favicon.ico' });
    
});
</script>
